\begin{frame}{Kết luận của đồ án}
\begin{itemize}
\item \textbf{Về mặt sản phẩm:}
\begin{itemize}
\item Xây dựng thành công hệ thống trợ lý pháp luật có khả năng chịu lỗi, dễ mở rộng.

\item Tích hợp LangGraph với 22 node trả lời chính xác, dẫn nguồn minh bạch.
\item Thanh toán tự động, xử lý tài liệu, hỗ trợ đàm thoại giọng nói.
\end{itemize}
\item \textbf{Về mặt công nghệ:}
\begin{itemize}

\item Tích hợp các giải pháp: Kong API Gateway, gRPC, Kafka, RabbitMQ
\item Đa dạng CSDL: Postgres, Vector DB (Pinecone/Qdrant/Milvus)
\item Object Storage lưu trữ file: Cloudflare R2, Supabase Bucket
\item Triển khai tự động: Terraform, Docker, K8s, GitOps, ArgoCD

\end{itemize}

\end{itemize}
