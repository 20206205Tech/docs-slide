\subsubsection{Thông tin các bảng trong CSDL của các dịch vụ}

Các dịch vụ đều có bảng quản lý việc di chuyển CSDL.
Trong python sử dụng alembic,
trong NestJS sử dụng typeorm.
Ngoài ra,
các dịch vụ đều có bảng
request\_logs
để ghi lại thông tin yêu cầu và phản hồi
của người dùng.

\begin{table}[H]
\centering
\begin{tabular}{|l|l|p{6cm}|}
\hline
\textbf{Cột} & \textbf{Kiểu dữ liệu} & \textbf{Mô tả} \\ \hline
version\_num & varchar & Mã phiên bản di chuyển CSDL. \\ \hline
\end{tabular}
\caption{Bảng \texttt{alembic\_version} quản lý phiên bản CSDL dịch vụ Python}
\end{table}

\begin{table}[H]
\centering
\begin{tabular}{|l|l|p{6cm}|}
\hline
\textbf{Cột} & \textbf{Kiểu dữ liệu} & \textbf{Mô tả} \\ \hline
id & serial & ID định danh của phiên bản migration. \\ \hline
timestamp & int8 & Thời điểm thực thi migration. \\ \hline
name & varchar & Tên file migration đã thực thi. \\ \hline
\end{tabular}
\caption{Bảng \texttt{migrations} quản lý phiên bản CSDL dịch vụ NestJS}
\end{table}

\begin{table}[H]
\centering
\begin{tabular}{|l|l|p{6cm}|}
\hline
\textbf{Cột} & \textbf{Kiểu dữ liệu} & \textbf{Mô tả} \\ \hline
id & uuid & Khóa chính của log yêu cầu HTTP. \\ \hline
requestId & varchar & Mã định danh duy nhất của yêu cầu. \\ \hline
method & varchar & Phương thức HTTP được sử dụng. \\ \hline
url & text & Đường dẫn của yêu cầu API. \\ \hline
statusCode & int4 & Mã trạng thái phản hồi HTTP. \\ \hline
requestPayload & jsonb & Dữ liệu chi tiết trong yêu cầu. \\ \hline
responsePayload & jsonb & Dữ liệu chi tiết hệ thống phản hồi. \\ \hline
processTime & float8 & Thời gian hệ thống xử lý yêu cầu. \\ \hline
created\_at & timestamptz & Thời gian ghi nhận log hệ thống. \\ \hline
clientIp & varchar & Địa chỉ IP của máy khách gửi yêu cầu. \\ \hline
\end{tabular}
\caption{Bảng \texttt{request\_logs} ghi nhận nhật ký yêu cầu và phản hồi API}
\end{table}
